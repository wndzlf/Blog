# GreedyBestChange

~~~ var M = 40
var c = [25,15,20,10,5,1]
var d = 6
var k = [0,0,0,0,0,0]

//GreedyBetterChange가 Correct 하기위한 c6의 값은?
func GreedyBetterChange(M:Int, c:[Int] , d:Int) -> [Int]{

var m = M
var changedK = [0,0,0,0,0,0]

for i in 0...d-1 {
k[i] = m / c[i]
m = m - c[i]*k[i]

}
changedK = k
return changedK
}
